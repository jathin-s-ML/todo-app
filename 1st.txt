package main

import (
	"bufio"
	"fmt"
	"os"
)

func main() {
	reader:=bufio.NewReader(os.Stdin)
	fmt.Println("TODO application")
	var n int
	fmt.Println("enter the number of task")
	fmt.Scan(&n)
	mp:=make(map[int]string)
	for i:=0;i<n;i++{
		fmt.Println("enter your task no : ",i+1)
		var str string
		str,_=reader.ReadString('\n')
		mp[i]=str;
	}
	fmt.Println("\n the list of task\n ")
	for key,value:=range mp{
		fmt.Println("key :",key,"value :",value)
	}
}
